    ```typescript
    import net from "net";

    export function allocatePort(): Promise<number> {
      return new Promise((resolve, reject) => {
        const server = net.createServer();
        server.listen(0, () => {
          const address = server.address();
          if (!address || typeof address === "string") {
            server.close();
            reject(new Error("Failed to allocate port"));
            return;
          }
          const port = address.port;
          server.close(() => resolve(port));
        });
        server.on("error", reject);
      });
    }
    ```

    **packages/db/src/embedded-postgres.ts:**
    ```typescript
    import { EmbeddedPostgres } from "embedded-postgres";
    import { allocatePort } from "./port-allocator.js";
    import fs from "fs";
    import path from "path";
    import os from "os";
    import { loadConfig, writeConfig } from "@leclaw/shared";

    const DEFAULT_DATA_DIR = path.join(os.homedir(), ".leclaw", "db");
    const DEFAULT_USER = "postgres";
    const DEFAULT_PASSWORD = "postgres";

    export interface DbConfig {
      dataDir?: string;
      user?: string;
      password?: string;
    }

    export async function initializeDb(config?: DbConfig): Promise<{
      connectionString: string;
      cleanup: () => Promise<void>;
    }> {
      const dataDir = config?.dataDir ?? DEFAULT_DATA_DIR;
      const user = config?.user ?? DEFAULT_USER;
      const password = config?.password ?? DEFAULT_PASSWORD;

      // Create data directory if not exists
      if (!fs.existsSync(dataDir)) {
        fs.mkdirSync(dataDir, { recursive: true });
      }

      const port = await allocatePort();

      const instance = new EmbeddedPostgres({
        databaseDir: dataDir,
        user,
        password,
        port,
        persistent: true,
        onLog: (msg) => console.log("[embedded-postgres]", msg),
        onError: (msg) => console.error("[embedded-postgres]", msg),
      });

      await instance.initialize();
      await instance.start();

      const connectionString = `postgres://${user}:${password}@127.0.0.1:${port}/postgres`;

      const cleanup = async () => {
        await instance.stop();
      };

      return { connectionString, cleanup };
    }

    export function createDb(connectionString: string) {
      // Returns a simple db client wrapper
      // In Phase 2 this will use actual database client
      return { connectionString, execute: async (sql: string) => sql };
    }
    ```
