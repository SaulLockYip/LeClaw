    ```typescript
    import { Command } from "commander";
    import path from "path";
    import os from "os";
    import fs from "fs";
    import { initializeDb } from "@leclaw/db";
    import { loadConfig, writeConfig, type LeClawConfig } from "@leclaw/shared";

    const CONFIG_DIR = path.join(os.homedir(), ".leclaw");
    const CONFIG_FILE = path.join(CONFIG_DIR, "config.json");
    const DB_DIR = path.join(CONFIG_DIR, "db");

    export function registerInitCommand(program: Command): void {
      program
        .command("init")
        .description("Initialize LeClaw configuration directory and database")
        .option("--dir <path>", "Custom config directory path")
        .action(async (opts) => {
          const configDir = opts.dir ? path.resolve(opts.dir) : CONFIG_DIR;
          const configFile = path.join(configDir, "config.json");
          const dbDir = path.join(configDir, "db");

          try {
            // Create directories
            if (!fs.existsSync(configDir)) {
              fs.mkdirSync(configDir, { recursive: true });
            }
            if (!fs.existsSync(dbDir)) {
              fs.mkdirSync(dbDir, { recursive: true });
            }

            // Check if config exists
            let config: LeClawConfig;
            if (fs.existsSync(configFile)) {
              config = loadConfig(configFile);
            } else {
              // Initialize embedded PostgreSQL
              process.stdout.write(JSON.stringify({ status: "initializing_db" }));
              const { connectionString, cleanup } = await initializeDb({ dataDir: dbDir });

              config = {
                version: "1.0.0",
                openclaw: { dir: "" },
                gateway: { url: "", key: "" },
                database: { connectionString },
              };

              await writeConfig(configFile, config);

              // Store cleanup reference for later use
              // In production, cleanup would be registered with process signal handlers
            }

            console.log(
              JSON.stringify({
                success: true,
                configDir,
                configFile,
                dbInitialized: true,
              })
            );
            process.exit(0);
          } catch (err) {
            console.error(
              JSON.stringify({
                success: false,
                error: err instanceof Error ? err.message : String(err),
                code: "INIT_ERROR",
              })
            );
            process.exit(1);
          }
        });
    }
    ```

    **Key behavior:**
    1. Creates ~/.leclaw/ directory if not exists (per CLI-01)
    2. Creates ~/.leclaw/db/ subdirectory
    3. Initializes embedded PostgreSQL using @leclaw/db (per DATA-01)
    4. Writes config with connection string to config.json
    5. Outputs structured JSON to stdout
    6. Outputs errors to stderr with error code
    7. Uses exit code 0 on success, 1 on failure (per D-05)

    ```typescript
    import { Command } from "commander";
    import path from "path";
    import os from "os";
    import fs from "fs";
    import { loadConfig } from "@leclaw/shared";

    const CONFIG_FILE = path.join(os.homedir(), ".leclaw", "config.json");

    interface StatusResult {
      openclaw: {
        configured: boolean;
        dir: string;
        exists: boolean;
        accessible: boolean;
      };
      gateway: {
        configured: boolean;
        url: string;
        reachable: boolean;
      };
      database: {
        configured: boolean;
        connectionString: string;
      };
      overall: "ok" | "warning" | "error";
    }

    export function registerStatusCommand(program: Command): void {
      program
        .command("status")
        .description("View connection status to OpenClaw and Gateway")
        .action(async () => {
          try {
            const result: StatusResult = {
              openclaw: {
                configured: false,
                dir: "",
                exists: false,
                accessible: false,
              },
              gateway: {
                configured: false,
                url: "",
                reachable: false,
              },
              database: {
                configured: false,
                connectionString: "",
              },
              overall: "ok",
            };

            // Load config
            if (fs.existsSync(CONFIG_FILE)) {
              const config = loadConfig(CONFIG_FILE);

              // Check OpenClaw directory
              if (config.openclaw?.dir) {
                result.openclaw.configured = true;
                result.openclaw.dir = config.openclaw.dir;
                result.openclaw.exists = fs.existsSync(config.openclaw.dir);
                if (result.openclaw.exists) {
                  try {
                    fs.accessSync(config.openclaw.dir, fs.constants.R_OK);
                    result.openclaw.accessible = true;
                  } catch {
                    result.openclaw.accessible = false;
                  }
                }
              }

              // Check Gateway
              if (config.gateway?.url) {
                result.gateway.configured = true;
                result.gateway.url = config.gateway.url;
                try {
                  const controller = new AbortController();
                  const timeout = setTimeout(() => controller.abort(), 5000);
                  const response = await fetch(config.gateway.url, {
                    method: "HEAD",
                    signal: controller.signal,
                  });
                  clearTimeout(timeout);
                  result.gateway.reachable = response.ok;
                } catch {
                  result.gateway.reachable = false;
                }
              }

              // Check Database
              if (config.database?.connectionString) {
                result.database.configured = true;
                result.database.connectionString = config.database.connectionString;
              }
            }

            // Determine overall status
            if (!result.database.configured) {
              result.overall = "error";
            } else if (!result.openclaw.configured || !result.gateway.configured) {
              result.overall = "warning";
            } else if (!result.openclaw.exists || !result.gateway.reachable) {
              result.overall = "warning";
            }

            console.log(JSON.stringify({ success: true, ...result }, null, 2));
            process.exit(result.overall === "error" ? 1 : 0);
          } catch (err) {
            console.error(
              JSON.stringify({
                success: false,
                error: err instanceof Error ? err.message : String(err),
                code: "STATUS_ERROR",
              })
            );
            process.exit(1);
          }
        });
    }
    ```

      "openclaw": { "configured": bool, "dir": string, "exists": bool, "accessible": bool },

      "database": { "configured": bool, "connectionString": string },

    ```

    import { Command } from "commander";
    import path from "path";
    import os from "os";
    import { fork } from "child_process";
    import { loadConfig } from "@leclaw/shared";

    const CONFIG_FILE = path.join(os.homedir(), ".leclaw", "config.json");

    export function registerStartCommand(program: Command): void {
      program
        .command("start")
        .description("Start LeClaw server (backend + web UI)")
        .option("--port <port>", "Server port", "3000")
        .option("--host <host>", "Server host", "0.0.0.0")
        .action(async (opts) => {
          try {
            // Load config to verify initialization
            if (!fs.existsSync(CONFIG_FILE)) {
              console.error(
                JSON.stringify({
                  success: false,
                  error: "Config not found. Run 'leclaw init' first.",
                  code: "NOT_INITIALIZED",
                })
              );
              process.exit(1);
            }

            const config = loadConfig(CONFIG_FILE);

            // Fork server process
            const serverPath = path.join(import.meta.dirname, "..", "..", "server", "dist", "index.js");
            const serverProcess = fork(serverPath, {
              env: {
                ...process.env,
                PORT: opts.port,
                HOST: opts.host,
                DATABASE_URL: config.database.connectionString,
              },
              stdio: ["inherit", "pipe", "pipe", "ipc"],
            });

            serverProcess.stdout?.on("data", (data) => {
              process.stdout.write(data);
            });

            serverProcess.stderr?.on("data", (data) => {
              process.stderr.write(data);
            });

            serverProcess.on("exit", (code) => {
              process.exit(code ?? 1);
            });

            // Handle Ctrl+C gracefully
            process.on("SIGINT", () => {
              serverProcess.kill("SIGINT");
            });
          } catch (err) {
            console.error(
              JSON.stringify({
                success: false,
                error: err instanceof Error ? err.message : String(err),
                code: "START_ERROR",
              })
            );
            process.exit(1);
          }
        });
    }
    ```

    import express, { Express } from "express";
    import cors from "cors";
    import { healthRouter } from "./routes/health.js";
    import { initializeDb } from "@leclaw/db";
    import { loadConfig } from "@leclaw/shared";
    import path from "path";
    import os from "os";

    export function createApp(): Express {
      const app = express();
      app.use(cors());
      app.use(express.json());
      app.use("/health", healthRouter);
      return app;
    }

    export async function startServer(): Promise<void> {
      const configPath = path.join(os.homedir(), ".leclaw", "config.json");
      const config = loadConfig(configPath);

      // Initialize DB if not already initialized
      const port = parseInt(process.env.PORT ?? "3000", 10);
      const host = process.env.HOST ?? "0.0.0.0";

      const app = createApp();
      app.listen(port, host, () => {
        console.log(
          JSON.stringify({
            success: true,
            server: { port, host },
            database: { connected: !!config.database.connectionString },
          })
        );
      });
    }
    ```

    import { Router, Request, Response } from "express";
    import { loadConfig } from "@leclaw/shared";
    import path from "path";
    import os from "os";

    export const healthRouter = Router();

    healthRouter.get("/", (req: Request, res: Response) => {
      const configPath = path.join(os.homedir(), ".leclaw", "config.json");
      let dbConnected = false;

      try {
        if (require("fs").existsSync(configPath)) {
          const config = loadConfig(configPath);
          dbConnected = !!config.database?.connectionString;
        }
      } catch {
        dbConnected = false;
      }

      res.json({
        status: "ok",
        timestamp: new Date().toISOString(),
        version: "1.0.0",
        database: dbConnected ? "connected" : "disconnected",
      });
    });
    ```
